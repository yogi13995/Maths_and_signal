\subsection{a}
\begin{frame}
\frametitle{Solution a)}
\footnotesize
\label{a}
from the $\Delta APB$ and $\Delta{AQB}$... 
\\
$$\norm{\vec{A-P}} = \norm{\vec{A-P}}$$
\\
$$\angle{AQB} = \angle{APB}$$
\\
$\vec{AB}$ is bisector of $\angle QAP$
\\
$$\implies\angle{AQB} = \angle{APB}$$
\\
thus from ASA conguransy
\\
$$\Delta APB \cong \Delta AQB$$
\end{frame}
