\begin{align}
\label{eq:constr_a}
x = \frac{\left(a^2 + c^2 - b^2\right)}{2*a}
\\
y = \sqrt{c^2 - x^2}
\\
\vec{A} &= \myvec{x \\ y}
\\
\vec{B} &= \myvec{0 \\ 0}
\\
\label{eq:constr_c}
\vec{C} &= \myvec{a  \\ 0 }
\\
\vec{P} &= \myvec{x+5 \\ y }

\end{align}
